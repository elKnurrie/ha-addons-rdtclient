#include <tunables/global>

profile rdt-client flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  capability net_bind_service,

  # Network access
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # File access
  /data/** rwk,
  /share/** rwk,
  /tmp/** rwk,
  /proc/sys/kernel/random/uuid r,
  
  # Application files
  /usr/** r,
  /app/** rk,
  /bin/** rix,
  /sbin/** rix,
  /lib/** rm,
  /lib64/** rm,
  
  # System files (read-only)
  /etc/** r,
  /var/lib/** r,
  
  # Deny dangerous paths
  deny /root/** rwklx,
  deny /sys/kernel/security/** rwklx,
}
