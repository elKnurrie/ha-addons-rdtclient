#include <tunables/global>

profile rdt-client flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  capability net_bind_service,
  capability chown,
  capability fowner,
  capability fsetid,
  capability setuid,
  capability setgid,

  # Network access
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # File access
  /data/** rwk,
  /share/** rwk,
  /tmp/** rwk,
  /proc/sys/kernel/random/uuid r,
  
  # Init and system executables
  /init rix,
  /usr/bin/s6-* rix,
  /command/** rix,
  /package/** rix,
  
  # Application files
  /usr/** rix,
  /app/** rix,
  /bin/** rix,
  /sbin/** rix,
  /lib/** rm,
  /lib64/** rm,
  
  # System files (read-only)
  /etc/** r,
  /var/lib/** r,
  /var/run/** rw,
  /run/** rw,
  
  # Deny dangerous paths
  deny /root/** rwklx,
  deny /sys/kernel/security/** rwklx,
}
