#include <tunables/global>

profile rdt-client flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Basic capabilities
  capability net_bind_service,
  capability chown,
  capability fowner,

  # Network access
  network inet stream,
  network inet6 stream,

  # File access
  /data/** rwk,
  /share/** rwk,
  /tmp/** rwk,
  
  # Application files
  /app/** rix,
  /usr/** rix,
  /bin/** rix,
  /lib/** rm,
  /lib64/** rm,
  
  # System files (read-only)
  /etc/** r,
  /proc/sys/kernel/random/uuid r,
  
  # Startup script
  /usr/local/bin/start.sh rix,
  
  # Deny sensitive paths
  deny /root/** rwklx,
  deny /sys/kernel/security/** rwklx,
}
