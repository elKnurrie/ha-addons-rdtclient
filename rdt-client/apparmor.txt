#include <tunables/global>

profile rdt-client flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  capability net_bind_service,
  capability chown,
  capability fowner,
  capability fsetid,
  capability setuid,
  capability setgid,
  capability sys_admin,

  # Network access
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # File access
  /data/** rwk,
  /share/** rwk,
  /tmp/** rwk,
  /proc/** r,
  /sys/** r,
  
  # Init and S6 overlay system - full access
  /init rix,
  /package/** rix,
  /command/** rix,
  /usr/bin/s6-* rix,
  /run/s6/** rix,
  /run/s6/basedir/bin/init rix,
  /var/run/s6/** rix,
  
  # Application files - full execute access
  /usr/** rix,
  /app/** rix,
  /bin/** rix,
  /sbin/** rix,
  /lib/** rix,
  /lib64/** rix,
  
  # System files
  /etc/** r,
  /var/lib/** r,
  /var/run/** rw,
  /run/** rwk,
  
  # Additional paths that might be needed
  /opt/** rix,
  /home/** rwk,
  
  # Deny dangerous paths (but allow what's needed)
  deny /root/.* rwklx,
  deny /sys/kernel/security/** rwklx,
}
